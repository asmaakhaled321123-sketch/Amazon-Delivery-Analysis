#include <windows.h>
#include <iostream>
#include <cstring>

struct ChatMemory {
    char message[256];
    bool newMessage;
};

int main() {
    // 1) Create or open shared memory
    HANDLE hMap = CreateFileMapping(
        INVALID_HANDLE_VALUE, NULL, PAGE_READWRITE,
        0, sizeof(ChatMemory), "Local\\ChatSharedMemory");

    ChatMemory* chat = (ChatMemory*)MapViewOfFile(
        hMap, FILE_MAP_ALL_ACCESS, 0, 0, sizeof(ChatMemory));

    // Initialize memory only once
    static bool initialized = false;
    if (!initialized) {
        chat->newMessage = false;
        initialized = true;
    }

    // 2) Create or open Mutex
    HANDLE mutex = CreateMutex(NULL, FALSE, "Local\\ChatMutex");

    std::cout << "Chat started. Type messages to send.\n";

    while (true) {
        // 3) Check for incoming message
        WaitForSingleObject(mutex, INFINITE);
        if (chat->newMessage) {
            std::cout << "\n[Peer]: " << chat->message << "\n> ";
            chat->newMessage = false;
        }
        ReleaseMutex(mutex);

        // 4) Send message
        std::string msg;
        std::cout << "> ";
        std::getline(std::cin, msg);

        WaitForSingleObject(mutex, INFINITE);
        strncpy(chat->message, msg.c_str(), 255);
        chat->newMessage = true;
        ReleaseMutex(mutex);
    }

    return 0;
}
